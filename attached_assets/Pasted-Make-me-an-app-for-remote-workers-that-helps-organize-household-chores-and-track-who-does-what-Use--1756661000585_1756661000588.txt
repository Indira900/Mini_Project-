class IntentClassifier:
    def _init_(self):
        self.intents = {
            "medication_question": ["when to take", "side effects", "dosage", "injection"],
            "emotional_support": ["sad", "worried", "anxious", "scared", "frustrated"],
            "procedure_inquiry": ["egg retrieval", "transfer", "ultrasound", "blood test"],
            "symptom_reporting": ["pain", "bleeding", "nausea", "bloating", "headache"],
            "appointment_help": ["schedule", "preparation", "what to expect"],
            "nutrition_advice": ["what to eat", "foods to avoid", "supplements"],
            "exercise_guidance": ["can I workout", "yoga", "walking", "rest"],
            "success_rate_question": ["chances", "probability", "success rate"]
        }
        
    def train_intent_classifier(self):
        # Training data for intent recognition
        training_texts = []
        training_labels = []
        
        for intent, keywords in self.intents.items():
            for keyword in keywords:
                training_texts.append(keyword)
                training_labels.append(intent)
        
        # TF-IDF Vectorization
        self.vectorizer = TfidfVectorizer()
        X = self.vectorizer.fit_transform(training_texts)
        
        # Naive Bayes Classifier
        self.classifier = MultinomialNB()
        self.classifier.fit(X, training_labels)
        
    def predict_intent(self, user_message):
        message_vector = self.vectorizer.transform([user_message])
        intent = self.classifier.predict(message_vector)[0]
        confidence = self.classifier.predict_proba(message_vector).max()
        return intent, confidence

class EmotionalStateAnalyzer:

        subjectivity = blob.sentiment.subjectivity  # 0 to 1

        }
        
        emotion_scores = {}
        for emotion, keywords in ivf_emotions.items():
            score = sum(1 for keyword in keywords if keyword in message.lower())

            "symptoms": ["bloating", "cramping", "nausea", "headache", "mood swings"],
            "body_parts": ["ovaries", "uterus", "fallopian tubes", "cervix"]
        }
        
    def extract_medical_info(self, text):
        doc = self.nlp(text)
        
        extracted_entities = {
            "medications": [],
            "procedures": [],
            "symptoms": [],
            "dates": [],
            "numbers": []
        }
        
        # Extract named entities
        for ent in doc.ents:
            if ent.label_ == "DATE":
                extracted_entities["dates"].append(ent.text)
            elif ent.label_ in ["CARDINAL", "QUANTITY"]:
                extracted_entities["numbers"].append(ent.text)
                
        # Extract custom medical entities
        text_lower = text.lower()
        for category, entities in self.medical_entities.items():
            for entity in entities:
                if entity.lower() in text_lower:
                    extracted_entities[category].append(entity)
                    
        return extracted_entities
